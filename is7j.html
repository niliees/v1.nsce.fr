// Setze den Motor an Pin P0
let motorPin = DigitalPin.P0
// Setze die LED an Pin P1
let ledPin = DigitalPin.P1
// Setze den externen Knopf an Pin P2
let buttonPin = DigitalPin.P2
// Setze den externen Knopf an Pin P2
// Überwache den Zustand des externen Knopfes
basic.forever(function () {
    if (pins.digitalReadPin(buttonPin) == 1) {
        // Motor einschalten
        pins.digitalWritePin(motorPin, 1)
        // LED einschalten
        pins.digitalWritePin(ledPin, 1)
        // Warte 60 Sekunden
        basic.pause(60000)
        // Motor und LED ausschalten
        pins.digitalWritePin(motorPin, 0)
        pins.digitalWritePin(ledPin, 0)
        // Warte, bis der Knopf losgelassen wird, um mehrfaches Auslösen zu verhindern
        while (pins.digitalReadPin(buttonPin) == 1) {
            // Kurze Pause, um CPU-Last zu reduzieren
            basic.pause(100)
        }
    }
})
